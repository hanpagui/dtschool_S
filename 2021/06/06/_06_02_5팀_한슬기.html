# > 소수는 1과 자기자신만으로 떨어지는 양의 정수 .
def is_prime(number):
    if number < 2:
        return False 

    for a in range(2,number):
        if number%a == 0:
            return False

    return True

def factorial(n):
    if n ==1:                 #n이 1일때 
        return 1              # 1을 반환 후 재귀호출 끝냄 
    print(n)
    return n * factorial(n -1)      #n과 팩토리얼 함수에 n-1을 넣어서 변환 

print(factorial(5))

def a_product():
    while True:
        product = input('상품명:')
        if product == '':
            break
        num = int(input('갯수:'))
        if product in inventory.keys():
            inventory[product] += num
        else:
            inventory[product] = num
    print('-'*50)
    for key,value in inventory.items():
        print(key,':',value)
